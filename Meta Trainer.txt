    function normalizeString(str) {
        if (!str) return "";
        return str.normalize("NFD").replace(/[\u0300-\u036f]/g, "").toLowerCase();
    }

    function updateState(newState) {
        let changed = false;
        for (const key in newState) {
            if (state[key] !== newState[key]) {
                state[key] = newState[key];
                changed = true;
            }
        }
        if (changed) renderUI();
    }

    function checkIsSingleplayer() {
        if (window.location.href.toLowerCase().includes('solo')) return true;
        return (document.body && document.body.innerHTML.toLowerCase().includes('solo'));
    }

    function init() {
        setupNetworkHooks();
        setupURLTracking();

        if (document.readyState === 'loading') {
            document.addEventListener('DOMContentLoaded', onReady);
        } else {
            onReady();
        }
    }

    function onReady() {
        updateState({ isSoloCached: checkIsSingleplayer() });
        setupIframeObserver();
        initUI();
    }

    function setupURLTracking() {
        let lastUrl = location.href;
        function handleUrlChange() {
            if (location.href !== lastUrl) {
                lastUrl = location.href;
                updateState({
                    currentLocation: null,
                    lastPano: null,
                    countryCode: null,
                    countryName: null,
                    regionName: null,
                    status: 'Waiting for a round...',
                    isSoloCached: checkIsSingleplayer()
                });
            }
        }

        window.addEventListener('popstate', handleUrlChange);
        const originalPushState = history.pushState;
        history.pushState = function(...args) {
            originalPushState.apply(this, args);
            handleUrlChange();
        };
        const originalReplaceState = history.replaceState;
        history.replaceState = function(...args) {
            originalReplaceState.apply(this, args);
            handleUrlChange();
        };
    }

    function getCountryFromCoords(lat, lng) {
        return new Promise((resolve) => {
            GM_xmlhttpRequest({
                method: 'GET',
                url: `https://nominatim.openstreetmap.org/reverse?lat=${lat}&lon=${lng}&format=json&accept-language=en&zoom=5`,
                headers: { 'User-Agent': 'MetaTrainer/4.6' },
                onload: function(response) {
                    try {
                        const data = JSON.parse(response.responseText);
                        if (data && data.address && data.address.country_code) {
                            resolve({
                                code: data.address.country_code.toLowerCase(),
                                name: data.address.country,
                                region: data.address.state || data.address.province || data.address.region || null
                            });
                        } else {
                            resolve({ code: 'unknown', name: 'Unknown Location', region: null });
                        }
                    } catch(e) {
                        resolve({ code: 'unknown', name: 'Error Resolving', region: null });
                    }
                },
                onerror: function() {
                    resolve({ code: 'unknown', name: 'Network Error', region: null });
                }
            });
        });
    }

    function extractCoordsFromText(text) {
        const panoMatch = text.match(/"pano(?:Id)?"\s*:\s*"([a-zA-Z0-9_\-]+)"/);
        if (panoMatch && panoMatch[1].length >= 10) return { pano: panoMatch[1] };

        const latMatch = text.match(/"lat(?:itude)?"\s*:\s*(-?\d+\.\d+)/);
        const lngMatch = text.match(/"l(?:ng|ongitude)"\s*:\s*(-?\d+\.\d+)/);

        if (latMatch && lngMatch) {
            const lat = parseFloat(latMatch[1]);
            const lng = parseFloat(lngMatch[1]);
            if (!isNaN(lat) && !isNaN(lng) && lat !== 0) return { lat, lng };
        }
        return null;
    }

    function setupNetworkHooks() {
        const OriginalWebSocket = window.WebSocket;
        window.WebSocket = function(url, protocols) {
            const ws = protocols ? new OriginalWebSocket(url, protocols) : new OriginalWebSocket(url);
            ws.addEventListener('message', function(event) {
                if (!state.isSoloCached) return;

                const dataStr = event.data;
                if (typeof dataStr === 'string' && (dataStr.includes('"lat"') || dataStr.includes('"pano"'))) {
                    let cleanStr = dataStr.replace(/^\d+/, '');
                    if (cleanStr) {
                        let coords = extractCoordsFromText(cleanStr);
                        if (!coords) {
                            try {
                                const data = JSON.parse(cleanStr);
                                coords = fastSearchForCoordinates(data);
                            } catch (e) {}
                        }
                        if (coords) handleFoundCoords(coords);
                    }
                }
            });
            return ws;
        };

        const originalFetch = window.fetch;
        window.fetch = async function(...args) {
            const response = await originalFetch.apply(window, args);
            if (!state.isSoloCached) return response;

            try {
                const url = typeof args[0] === 'string' ? args[0] : (args[0]?.url || '');
                if (!url.includes('.jpg') && !url.includes('.png')) {
                    const clone = response.clone();
                    clone.text().then(text => {
                        if (text.includes('"lat"') || text.includes('"pano"')) {
                            let coords = extractCoordsFromText(text);
                            if (!coords) {
                                try {
                                    const data = JSON.parse(text);
                                    coords = fastSearchForCoordinates(data);
                                } catch(e){}
                            }
                            if (coords) handleFoundCoords(coords);
                        }
                    }).catch(() => {});
                }
            } catch(e) {}
            return response;
        };
    }

    function checkIframeSrc(src) {
        if (!src || (!src.includes('google') && !src.includes('streetview'))) return;

        const coordsMatch = src.match(/[cl]b?ll=(-?\d+\.\d+)[,%](-?\d+\.\d+)/) ||
                            src.match(/location=(-?\d+\.\d+)[,%](-?\d+\.\d+)/);
        if (coordsMatch) {
            handleFoundCoords({lat: parseFloat(coordsMatch[1]), lng: parseFloat(coordsMatch[2])});
            return;
        }

        const panoMatch = src.match(/pano=([a-zA-Z0-9_\-]+)/);
        if (panoMatch) {
            const pano = panoMatch[1];
            if (state.lastPano !== pano) {
                updateState({ lastPano: pano });
                handleFoundCoords({ pano });
            }
        }
    }

    function setupIframeObserver() {
        const iframeObserver = new MutationObserver((mutations) => {
            if (!state.isSoloCached) return;
            for (const mutation of mutations) {
                if (mutation.type === 'childList') {
                    mutation.addedNodes.forEach(node => {
                        if (node.tagName === 'IFRAME') checkIframeSrc(node.src);
                        else if (node.querySelectorAll) {
                            node.querySelectorAll('iframe').forEach(ifr => checkIframeSrc(ifr.src));
                        }
                    });
                } else if (mutation.type === 'attributes' && mutation.target.tagName === 'IFRAME') {
                    checkIframeSrc(mutation.target.src);
                }
            }
        });

        iframeObserver.observe(document.documentElement, {
            childList: true,
            subtree: true,
            attributes: true,
            attributeFilter: ['src']
        });

        document.querySelectorAll('iframe').forEach(ifr => checkIframeSrc(ifr.src));
    }

    async function getMetadata(pano) {
        if (state.panoCache.has(pano)) return state.panoCache.get(pano);
        try {
            let type = 2;
            if (String(pano).startsWith('CIHM') || pano.length != 22) type = 10;
            const payload = JSON.stringify([["apiv3"], ["en", "US"], [[[type, pano]]], [[1, 2, 3, 4, 8, 6]]]);

            const response = await fetch(`https://maps.googleapis.com/$rpc/google.internal.maps.mapsjs.v1.MapsJsInternalService/GetMetadata`, {
                method: "POST",
                headers: { "content-type": "application/json+protobuf" },
                body: payload,
                mode: "cors", credentials: "omit"
            });

            if (response.ok) {
                const data = await response.json();
                const lat = data?.[1]?.[0]?.[5]?.[0]?.[1]?.[0]?.[2];
                const lng = data?.[1]?.[0]?.[5]?.[0]?.[1]?.[0]?.[3];
                if (lat !== undefined && lng !== undefined) {
                    const coords = {lat, lng};
                    state.panoCache.set(pano, coords);
                    return coords;
                }
            }
        } catch (e) {}
        return null;
    }

    function fastSearchForCoordinates(obj, depth = 0) {
        if (!obj || typeof obj !== 'object' || depth > 4) return null;

        if (obj.lat !== undefined && obj.lng !== undefined) {
            const lat = parseFloat(obj.lat), lng = parseFloat(obj.lng);
            if (!isNaN(lat) && !isNaN(lng) && lat !== 0) return { lat, lng };
        }
        if (obj.latitude !== undefined && obj.longitude !== undefined) {
            const lat = parseFloat(obj.latitude), lng = parseFloat(obj.longitude);
            if (!isNaN(lat) && !isNaN(lng) && lat !== 0) return { lat, lng };
        }
        if (obj.panoId) return { pano: obj.panoId };
        if (obj.pano && typeof obj.pano === 'string' && obj.pano.length >= 10) return { pano: obj.pano };

        const proto = Object.getPrototypeOf(obj);
        if (Array.isArray(obj) || proto === Object.prototype || proto === null) {
            for (const key in obj) {
                const val = obj[key];
                if (val && typeof val === 'object') {
                    const res = fastSearchForCoordinates(val, depth + 1);
                    if (res) return res;
                }
            }
        }
        return null;
    }

    async function handleFoundCoords(result) {
        if (!state.isSoloCached || !result) return;

        if (result.pano) {
            updateState({ status: 'Fetching location...' });
            const coords = await getMetadata(result.pano);
            if (coords) setCoords(coords.lat, coords.lng);
        } else if (result.lat !== undefined && result.lng !== undefined) {
            setCoords(result.lat, result.lng);
        }
    }

    async function setCoords(lat, lng) {
        const hasChanged = !state.currentLocation ||
            Math.abs(state.currentLocation.lat - lat) > 0.0001 ||
            Math.abs(state.currentLocation.lng - lng) > 0.0001;

        if (hasChanged) {
            updateState({
                currentLocation: {lat, lng},
                countryCode: null,
                countryName: null,
                regionName: null,
                status: "Analyzing landscape..."
            });

            const geoData = await getCountryFromCoords(lat, lng);
            updateState({
                countryCode: geoData.code,
                countryName: geoData.name,
                regionName: geoData.region,
                status: "Location acquired"
            });
        }
    }

    function initUI() {
        overlayElement = document.createElement('div');
        overlayElement.id = 'meta-trainer-ui';

        overlayElement.innerHTML = `
            <div id="meta-trainer-header" style="cursor:move;">
                <span style="display: flex; align-items: center; gap: 8px;">
                    <span class="meta-trainer-lock" id="mt-lock-icon" style="cursor:pointer; font-size:14px;" title="Lock position">🔓</span>
                    <span style="display:flex; align-items:center; gap:6px;">
                        <span id="mt-status-dot" style="width:8px; height:8px; border-radius:50%; background:#ffe562; box-shadow: 0 0 6px #ffe562;"></span>
                        Meta Trainer
                    </span>
                </span>
                <span style="font-size:11px; opacity:0.6; font-weight: 400;">[Click to collapse]</span>
            </div>
            <div class="meta-trainer-content" id="mt-content-area" data-render-key="" data-mode="none"></div>
            <div class="meta-trainer-resizer" title="Drag to resize"></div>
        `;
        document.body.appendChild(overlayElement);

        overlayElement.addEventListener('mousedown', (e) => {
            if (e.target.closest('.meta-trainer-resizer')) {
                let startY = e.clientY;
                let startHeight = overlayElement.offsetHeight;
                overlayElement.style.maxHeight = 'none';

                function onMouseMoveResize(moveEvent) {
                    overlayElement.style.height = Math.max(100, startHeight + (moveEvent.clientY - startY)) + 'px';
                }
                function onMouseUpResize() {
                    window.removeEventListener('mousemove', onMouseMoveResize);
                    window.removeEventListener('mouseup', onMouseUpResize);
                    window.removeEventListener('blur', onMouseUpResize);
                }
                window.addEventListener('mousemove', onMouseMoveResize);
                window.addEventListener('mouseup', onMouseUpResize);
                window.addEventListener('blur', onMouseUpResize);
                e.preventDefault();
                e.stopPropagation();
                return;
            }

            if (state.uiLocked) return;
            if (!e.target.closest('#meta-trainer-header')) return;
            if (e.target.closest('.meta-trainer-lock')) return;

            isDraggingFlag = false;
            let startX = e.clientX;
            let startY = e.clientY;
            let initialX = overlayElement.offsetLeft;
            let initialY = overlayElement.offsetTop;

            overlayElement.style.right = 'auto';
            overlayElement.style.bottom = 'auto';

            function onMouseMove(moveEvent) {
                let dx = moveEvent.clientX - startX;
                let dy = moveEvent.clientY - startY;
                if (Math.abs(dx) > 3 || Math.abs(dy) > 3) isDraggingFlag = true;
                overlayElement.style.left = (initialX + dx) + 'px';
                overlayElement.style.top = (initialY + dy) + 'px';
            }

            function onMouseUp() {
                window.removeEventListener('mousemove', onMouseMove);
                window.removeEventListener('mouseup', onMouseUp);
                window.removeEventListener('blur', onMouseUp);
                setTimeout(() => { isDraggingFlag = false; }, 50);
            }

            window.addEventListener('mousemove', onMouseMove);
            window.addEventListener('mouseup', onMouseUp);
            window.addEventListener('blur', onMouseUp);
        });

        overlayElement.addEventListener('click', (e) => {
            const lockBtn = e.target.closest('.meta-trainer-lock');
            if (lockBtn) {
                updateState({ uiLocked: !state.uiLocked });
                return;
            }

            if (!isDraggingFlag && e.target.closest('#meta-trainer-header')) {
                overlayElement.classList.toggle('minimized');
            }
        });

        renderUI();
    }

    function renderUI() {
        if (!overlayElement) return;

        if (!state.isSoloCached) {
            overlayElement.style.display = 'none';
            return;
        }

        overlayElement.style.display = 'flex';

        const lockIconEl = document.getElementById('mt-lock-icon');
        const statusDotEl = document.getElementById('mt-status-dot');
        const contentAreaEl = document.getElementById('mt-content-area');
        const headerEl = document.getElementById('meta-trainer-header');

        if (!lockIconEl || !statusDotEl || !contentAreaEl || !headerEl) return;

        lockIconEl.textContent = state.uiLocked ? '🔒' : '🔓';
        lockIconEl.title = state.uiLocked ? 'Unlock dragging' : 'Lock position';
        headerEl.style.cursor = state.uiLocked ? 'pointer' : 'move';

        const statusColor = state.countryCode ? '#6deec0' : '#ffe562';
        statusDotEl.style.background = statusColor;
        statusDotEl.style.boxShadow = `0 0 6px ${statusColor}`;

        if (state.countryCode) {
            const currentRenderKey = `c_${state.countryCode}_${state.regionName}`;

            if (contentAreaEl.dataset.renderKey !== currentRenderKey) {
                const dbEntry = TIPS_DB[state.countryCode];
                const countryDisplayName = dbEntry ? dbEntry.name : state.countryName;
                let tipsHTML = '';

                if (dbEntry && dbEntry.data) {
                    tipsHTML = `<ul class="mt-list">
                        ${dbEntry.data.map((val, i) => `<li style="margin-bottom: 6px;"><b>${CATEGORIES[i]}:</b> ${val}</li>`).join('')}
                    </ul>`;
                } else {
                    tipsHTML = `<p style="font-size:13px; color:rgba(255, 255, 255, 0.7); margin: 8px 0 0 0; font-style: italic;">No 18-point meta loaded for this country yet. Use general clues.</p>`;
                }

                // ALWAYS display region/state info if nominatim gave us one
                if (state.regionName) {
                    let actualRegionName = state.regionName;
                    let matchedObj = null;

                    if (dbEntry && dbEntry.normalizedRegions) {
                        const normRegion = normalizeString(state.regionName);
                        matchedObj = dbEntry.normalizedRegions.find(r =>
                            normRegion.includes(r.normalized) || r.normalized.includes(normRegion)
                        );
                    }

                    if (matchedObj) actualRegionName = matchedObj.original;

                    tipsHTML += `
                        <div class="mt-subtitle" style="color: #6deec0; margin-top: 16px;">Region Detected: ${actualRegionName}</div>
                    `;

                    if (matchedObj && dbEntry.regions[actualRegionName]) {
                        tipsHTML += `
                        <ul class="mt-list">
                            ${dbEntry.regions[actualRegionName].map(t => `<li style="margin-bottom: 4px;">${t}</li>`).join('')}
                        </ul>`;
                    } else {
                        tipsHTML += `
                        <ul class="mt-list">
                            <li style="margin-bottom: 4px; color: rgba(255,255,255,0.6); font-style: italic;">No specific meta loaded for this region. General country clues apply.</li>
                        </ul>`;
                    }
                }

                contentAreaEl.innerHTML = `
                    <div class="mt-subtitle mt-accent-green">Current Region</div>
                    <div class="mt-title" style="margin-bottom: 12px;">${countryDisplayName}</div>
                    <div style="border-top: 1px solid rgba(255,255,255,0.1); padding-top: 12px;">
                        <div class="mt-subtitle mt-accent-yellow">Visual Metas</div>
                        ${tipsHTML}
                    </div>
                `;
                contentAreaEl.dataset.renderKey = currentRenderKey;
                contentAreaEl.dataset.mode = 'country';
            }
        } else {
            if (contentAreaEl.dataset.mode !== 'status') {
                contentAreaEl.innerHTML = `
                    <div class="mt-subtitle mt-accent-yellow">Status</div>
                    <div id="mt-status-text" style="font-size: 14px; color: #dbecff; margin-top: 5px;">${state.status}</div>
                `;
                contentAreaEl.dataset.mode = 'status';
                contentAreaEl.dataset.renderKey = '';
            } else {
                const statusTextEl = document.getElementById('mt-status-text');
                if (statusTextEl && statusTextEl.textContent !== state.status) {
                    statusTextEl.textContent = state.status;
                }
            }
        }
    }
